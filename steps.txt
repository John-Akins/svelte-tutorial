1. Install / download svelte file
	- npx degit sveltejs/template my-svelte-project
2. install css bundle plugin
	- npm install rollup-plugin-css-only -D
	@ rollup.config.js
	* import css from 'rollup-plugin-css-only'; 
	* add css to plugins 
	 css({ output: 'plugin/extra.css' }),

3. install materialize
	- npm install materialize-css@next 

4. link materialize css and js in app
	@ App.svelte
	* import '..node_modules/materialize.css/dist/css/maerialize.min.css'
	* import '..node_modules/materialize.js/dist/js/maerialize.min.js'

5. ...
